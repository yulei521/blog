一道简单的JavaScript编程题

发表于 2016-10-10   |   分类于 JavaScript   |     |   阅读次数

之前做的一道关于JavaScript的编程题，现在拿出来尝试用多种方法来解决下。题目如下： 

var obj = {1: "Hello", 2: "my", 3: "beautiful", 4: "life!"}
试用console.log() 输出结果 Hello my beautiful life!
1、 直接利用for in

首先想到的肯定是通过for in 获取属性值然后拼接，代码如下：

var obj = {1: "Hello", 2: "my", 3: "beautiful", 4: "life!"};
function conobj(obj){
    var string = "";
    for(var item in obj){
        string+=obj[item] + " ";
    }
    console.log(string);
 }
conobj(obj);
2、 给Object添加方法

或者给Object添加个方法，然后在方法中对属性值进行拼接，代码类似，如下：

var obj = {1: "Hello", 2: "my", 3: "beautiful", 4: "life!"};
Object.prototype.conobj = function(){
    var str = "";
    for(var item in this){
        if(typeof this[item] == "string"){
            str += this[item] + " ";
        } 
    }
console.log(str);
}    
obj.conobj();

上述两种方法类似，都是通过for in 获取对象属性，然后通过属性获取属性值进行字符串拼接，记得加空格，不同的是第二种方法对this对象进行for循环时需要过滤下，不然会把对象的函数作为值进行拼接。

3、 转换为字符串进行分割

或者先把对象转换为字符串利用JSON.stringify，然后利用多次的split分割出属性值，进行拼接，相比上两种有点复杂，代码如下：

var obj = {1: "Hello", 2: "my", 3: "beautiful", 4: "life!"};
function conobj(obj){
    var arr2str = JSON.stringify(obj);
    var arr = arr2str.slice(1, arr2str.length-1).split(",");
    var string ="";
    for(var i=0;i<arr.length;i++){
        var temp = arr[i].split(":")[1];
        string+= temp.slice(1, temp.length-1)+ " ";
    }
console.log(string);
}
conobj(obj);
4、 利用正则匹配

构造出正则表达式使用全局模式匹配找出对象属性值连接起来，具体代码如下：

var obj = {1: "Hello", 2: "my", 3: "beautiful", 4: "life!"};
function conobj(obj){
    var str = JSON.stringify(obj);
    var pattern = /"\d{1}":\s*"([a-zA-Z]+!?)"/gi;
    var string = "";
    var matches;
    var i = 4;
    while(i>0){
        matches = pattern.exec(str);
        string += matches[1] + ' ';
        i--;
    }
    console.log(string);
 };
 conobj(obj);
